Portals

Extag内置Portal和Portal.Target组件，用以将定义在父组件模板中的部分内容渲染到父组件以外的地方。Portal组件的target属性对应Portal.Target的name，这两个属性不支持动态修改。

class BoxA extends Extag.Component {
  static get template() {
    return `
      <div>
        box-a
        <Portal target="b">
            <div class="msg">I'm from box-a</div>
        </Portal>
        <Portal.Target name="a"></Portal.Target>
      </div>
    `
  }
  static get resources() {
    return {
      Portal: Extag.Portal
    }
  }
}
class BoxB extends Extag.Component {
  static get template() {
    return `
      <div>
        box-b
        <Portal target="a">
            <div class="msg">I'm from box-b</div>
        </Portal>
        <Portal.Target name="b"></Portal.Target>
      </div>
    `
  }
  static get resources() {
    return {
      Portal: Extag.Portal
    }
  }
}
(new BoxA()).attach(ExtagDOM.query('#box-a'));
(new BoxB()).attach(ExtagDOM.query('#box-b'));
